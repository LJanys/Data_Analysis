usquared<-function(u,k,l)
{
  print("This loop calculates the square of the first k elements of vector u1, for which u>0.8 ")
  # Initialize `usq`
  usq <- 0
  for(i in 1:k) {
    if(abs(u[i])>l){
      # i-th element of `u1` squared into `i`-th position of `usq`
      usq[i] <- u[i]*u[i]
    }
    else{
      usq[i]<-u[i]
    }
  }
  z=sum(abs(u[1:k])>l)
  return=list(usq=usq,z=z)
}
